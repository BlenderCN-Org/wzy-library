class Queue{public:	Type MaxValue(Type x, Type y)	{		If(x > y)			return x;		else			return y;	}	Type Queue::Max()	{		return MaxValue(stackA.Max(), stackB.Max());	}	Insert2Queue(v)	{		stackB.push(v);	}	Type DeQueue()	{		If(stackA.empty())		{			While(!stackB.empty())				stackA.push(stackB.pop())		}		return stackA.pop();	}private:	stack stackA;	stack stackB;}
